// Custom Skeleton Component with Reanimated import Animated, { useSharedValue, withRepeat, withTiming, useAnimatedStyle } from 'react-native-reanimated'; const Skeleton = ({width, height, borderRadius = 8}) => { const shimmerX = useSharedValue(-200); React.useEffect(() => { shimmerX.value = withRepeat(withTiming(200, {duration: 1500}), -1); }, []); const animatedStyle = useAnimatedStyle(() => ({ transform: [{translateX: shimmerX.value}] })); return ( <View style={{width, height, borderRadius, backgroundColor: 'rgba(255,255,255,0.05)', overflow: 'hidden'}}> <Animated.View style={[{width: '100%', height: '100%', backgroundColor: 'rgba(255,255,255,0.1)'}, animatedStyle]} /> </View> ); };
