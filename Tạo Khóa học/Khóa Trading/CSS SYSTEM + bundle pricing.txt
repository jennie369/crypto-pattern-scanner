**Vanilla JavaScript (IIFE Pattern):**
```html
<script>
(function() {
    'use strict';
    
    // Wait for DOM to be ready
    function ready(fn) {
        if (document.readyState !== 'loading') {
            fn();
        } else {
            document.addEventListener('DOMContentLoaded', fn);
        }
    }
    
    // Initialize quiz when DOM is ready
    ready(function() {
        initQuiz();
```
